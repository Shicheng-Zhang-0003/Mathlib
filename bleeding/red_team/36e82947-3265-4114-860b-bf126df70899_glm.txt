### 2. The Masterpiece: `ml_trig.h`
In v10, `ml_sin` was a dirty `#define` macro in `profiles.h` that pointed to different functions. In v11A1, `ml_sin` is a *real function*, and the profile routing happens cleanly inside it.

```c
// ml_trig.h
#ifndef MATHLIB_ML_TRIG_H
#define MATHLIB_ML_TRIG_H

#include "ml_core.h"
#include "internal/minimax.h"
#include "internal/payne_hanek.h"
#include "internal/cordic.h"

// ---------------------------------------------------------
// Profile-Routed Trigonometry (The v11 Way)
// ---------------------------------------------------------

static inline double ml_sin(double x) {
#if defined(MATHLIB_PROFILE_EMBEDDED)
    // Embedded uses CORDIC, but we still need to handle NaN/Inf before passing it down
    if (ml_isnan(x) || ml_isinf(x)) return 0.0/0.0;
    double s, c;
    ml_cordic_sincos(x, &s, &c);
    return s;
#else
    // SCIENTIFIC and GRAPHICS both use the high-precision Minimax polynomial.
    // GRAPHICS gets its speed from SIMD batching later in the pipeline, not from
    // degrading single-value accuracy.
    return ml_minimax_sin(x);
#endif
}

static inline double ml_cos(double x) {
    if (ml_isnan(x) || ml_isinf(x)) return 0.0/0.0;
#if defined(MATHLIB_PROFILE_EMBEDDED)
    double s, c;
    ml_cordic_sincos(x, &s, &c);
    return c;
#else
    return ml_minimax_cos(x);
#endif
}

// True tangent implemented properly (avoiding division by zero from cos)
static inline double ml_tan(double x) {
    double s = ml_sin(x);
    double c = ml_cos(x);
    if (c == 0.0) return (s > 0) ? 1.0/0.0 : -1.0/0.0; // Return correct signed Inf
    return s / c;
}

#endif
```

### 3. The Clean `ml_exp_log.h`
Same philosophy. Kill the unprefixed `exponential()`. Make `ml_exp` the undeniable king.

```c
// ml_exp_log.h
#ifndef MATHLIB_ML_EXP_LOG_H
#define MATHLIB_ML_EXP_LOG_H

#include "ml_core.h"

static inline double ml_exp(double x) {
    if (x == 0.0) return 1.0;
    if (x > 709.78) return 1.0 / 0.0;
    if (x < -745.13) return 0.0;

    // Cody-Waite range reduction
    double n = ml_round(x / 0.693147180559945309417);
    double r = x - n * 0.69314718036912381649 - n * 1.90821490974462528503e-10;

    // 20th-degree Taylor
    double term = 1.0, result = 1.0;
    for (int i = 1; i <= 20; i++) {
        term *= r / i;
        result += term;
    }
    return ml_ldexp_pure(result, (int)n);
}

static inline double ml_log(double x) {
    if (x == 0.0) return -1.0 / 0.0;
    if (x < 0.0) return 0.0 / 0.0;
    if (x == 1.0) return 0.0;

    int e;
    double m = ml_frexp_pure(x, &e);
    if (m < 0.7071067811865475) { m *= 2.0; e--; }

    double z = (m - 1.0) / (m + 1.0);
    double z2 = z * z;
    double result = z, term = z;
    for (int i = 3; i <= 21; i += 2) {
        term *= z2;
        result += term / i;
    }
    return 2.0 * result + e * 0.693147180559945309417;
}

static inline double ml_pow(double x, double y) { return ml_exp(y * ml_log(x)); }

#endif
```
